redis-cli
-c 可以实现自动定位到新的节点去获得相应的值(解决(error) MOVED 问题)
redis-cli -c

如果不知道在哪儿可以查找下
whereis redis-cli

/usr/bin/redis-cli
2* 显示所有key
keys *

3* 获取key值
get keyname

4* 设置key的value

set key value 

set cp:mode 3

5* 放数据 Lpush 命令将一个或多个值插入到列表头部

lpush key value

6* 删除key
del key
del cp:error

7* 输入密码
auth 密码

8* 正常关闭redis
redis-cli
(如果有密码需要先输入密码)
auth 密码
shutdown

9* 查看redis版本
redis-cli --version

10* 清空整个 Redis 服务器的数据
flushall

11* 清空当前库中的所有 key
flushdb

12* 模糊查询key
*任意多个
?单个
[]括号内的1个

keys *
keys token:*
keys token:???????????

13* 查看key超时时间
ttl key

14* 正常停止redis
redis-cli shutdown
redis-cli -a 密码  shutdown

15* 对缓存使用分组
使用:分隔即可，如下
jat:system:id:apiId

/opt/redis-5.0.7/bin/redis-cli --cluster info 10.0.192.31:6379 -a "yourpassword1"

SELECT index

redis安装 tar包版本
tar -zxvf redis-5.0.7.tar.gz
rm -rf redis-5.0.7.tar.gz
mv redis-5.0.7/ redis

cd redis
make MALLOC=libc
make install PREFIX=/opt/redis-5.0.7/
rm -rf /opt/redis

/opt/redis-5.0.7/bin/redis-server /data/redis/redis6379/config/redis.conf

nohup /opt/redis-5.0.7/bin/redis-server /data/redis/redis6379/config/redis.conf >/dev/null 2>&1 &

安全停止集群脚本
stop-cluster.bat

@echo off
chcp 65001 >nul
setlocal

echo ========================================
echo    Redis Cluster 安全关闭脚本
echo ========================================

set REDIS_CLI=redis-cli.exe
set PASSWORD=yourpassword

for %%p in (6379 6380 6381 6382 6383 6384) do (
    echo 正在关闭端口 %%p ...
    %REDIS_CLI% -p %%p -a "%PASSWORD%" shutdown >nul 2>&1
    if %errorlevel% equ 0 (
        echo [OK]
    ) else (
        echo [FAIL] - 节点可能已停止
    )
    timeout /t 1 /nobreak >nul
)

echo ========================================
echo    所有关闭命令已发送
echo ========================================
pause
