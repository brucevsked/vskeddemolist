48* 批量重命名脚本
@echo ON
rem 批量重命名能够将本目录下jpg文件中-替换为_

setlocal enabledelayedexpansion
for /f "delims=" %%a in ('dir /b *.jpg') do (
    echo %%a
    set file="%%a"
    set file="!file:-=_!"
    ren "%%a" "!file!"
)

48* 批量重命名脚本重命名计数+1
@echo ON
rem 批量重命名能够将本目录下jpg文件按序列命名

setlocal enabledelayedexpansion
set count=0
for /f "delims=" %%a in ('dir /b *.jpg') do (
    set /a count = !count! + 1
    ren "%%a" "!count!.jpg"
)

48* 去掉文件名中空格

@echo ON
rem 批量重命名能够去掉当前目录文件中空格

setlocal enabledelayedexpansion
for /f "delims=" %%a in ('dir /b') do (
    echo %%a
    set file="%%a"
    set file="!file: =!"
    ren "%%a" "!file!"
)
